float liczba1 = 0.0, liczba2 = 0.0, wynik = 0.0;  //zadeklarowanie zmiennych
int opcja = 0;

void setup() 
{
	Serial.begin(9600);
}

void loop()				//wyswietlenie informacji na temat operacji
{
	Serial.println("Kalkulator prosty do liczb dwucyfrowych");
	Serial.println("1 - dodawanie"); 
	Serial.println("2 - odejmowanie"); 
	Serial.println("3 - mnozenie"); 
	Serial.println("4 - dzielenie");
	Serial.println("wybierz liczbe odpowiadajaca dzialaniu, ktore chcesz wykonac");
	while(Serial.available() == 0)		//oczekiwanie na wybor operacji
	{
	}
	opcja=Serial.parseInt();
	Serial.println(opcja);
	if(opcja<1 || opcja>4)	//sprawdzenie czy dobrze wprowadzono liczbe odpowiadajaca dzialaniu
	{
		Serial.println("Wybrano zly numer, sprobuj jeszcze raz");
	}
	Serial.println("Podaj pierwsza liczbe: ");
	while(Serial.available == 0);		//oczekiwanie na wprowadzenie 1 liczby
	liczba1 = Serial.parseFloat();
	Serial.println(liczba1);
	
	Serial.println("Podaj druga liczbe: ");
	while(Serial.available == 0); 		//oczekiwanie na wprowadzenie 2 liczby
	liczba2 = Serial.parseFloat();		
	Serial.println(liczba2);

	switch(opcja)
	{
		case 1:							//dodawanie
		if(abs(liczba1)<100 || abs(liczba2)<100)		//sprawdzenie czy wprowadzono liczbe dwucyfrowa
		{
			wynik=liczba1+liczba2;
			Serial.println("Dzialanie");
			Serial.println(liczba1);
			Serial.println("+");
			Serial.println(liczba2);
			Serial.println("=");
			Serial.println(wynik);
		}
		else
		{
			Serial.println("Nie podano liczby dwucyfrowej!");
		}
		break;

		
		case 2:							//odejmowanie
		if(abs(liczba1)<100 || abs(liczba2)<100)
		{
			wynik=liczba1-liczba2;
			Serial.println("Dzialanie");
			Serial.println(liczba1);
			Serial.println("-");
			Serial.println(liczba2);
			Serial.println("=");
			Serial.println(wynik);
		}
		else
		{
			Serial.println("Nie podano liczby dwucyfrowej!");
		}
		break;


		case 3:							//mnozenie
		if(abs(liczba1)<100 || abs(liczba2)<100)		//sprawdzenie czy wpisane sa liczby dwucyfrowe
		{
			wynik=liczba1*liczba2;
			Serial.println("Dzialanie");
			Serial.println(liczba1);
			Serial.println("*");
			Serial.println(liczba2);
			Serial.println("=");
			Serial.println(wynik);
		}
		else
		{
			Serial.println("Nie podano liczby dwucyfrowej!");
		}
		break;


		case 4:							//dzielenie
		if(liczba2 == 0)
		{
			Serial.println("Musisz powtorzyc wiedze z podstawowki :)");	//sprawdzenie czy nie dzielimy przez 0
		}
		else if(abs(liczba1)<100 || abs(liczba2)<100)
		{
			wynik=liczba1/liczba2;
			Serial.println("Dzialanie:");
			Serial.println(liczba1);
			Serial.println("/");
			Serial.println(liczba2);
			Serial.println("=");
			Serial.println(wynik);
		}
		 else
		{
			Serial.println("Nie podano liczby dwucyfrowej");
		}
		break;
	}
}
